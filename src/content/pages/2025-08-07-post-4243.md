Generate Neon Key


document.addEventListener('DOMContentLoaded', function() {
  const keyOutput = document.getElementById('key-output');
  const btn = document.getElementById('generate-btn');

  function generateKeySegment(length) {
    const chars = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789';
    let segment = '';
    for (let i = 0; i < length; i++) {
      segment += chars.charAt(Math.floor(Math.random() * chars.length));
    }
    return segment;
  }

  function generateKey() {
    const key = `${generateKeySegment(4)}-${generateKeySegment(4)}-${generateKeySegment(4)}-${generateKeySegment(4)}`;
    keyOutput.textContent = key;
  }

  btn.addEventListener('click', () => {
    generateKey();
    // Brief flash effect on click
    keyOutput.style.color = '#fff';
    setTimeout(() => {
      keyOutput.style.color = '#0ff';
    }, 150);
  });

  // Generate initial key on page load
  generateKey();
});
