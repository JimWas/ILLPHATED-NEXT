  // Kick autoplay-with-sound fallback
  (function(){
    const iframe = document.getElementById('kick-player');
    const btn = document.getElementById('kick-unmute-btn');
    const audio = document.createElement('audio');
    audio.src = "data:audio/mp3;base64,//uQZAAAAAAAAAAAAAAAAAAAA"; // silent clip placeholder
    audio.autoplay = true;
    audio.muted = false;

    const playPromise = audio.play && audio.play();
    if (playPromise && typeof playPromise.then === 'function') {
      playPromise.catch(() => {
        btn.style.display = 'inline-block';
      });
    } else {
      btn.style.display = 'inline-block';
    }

    btn.addEventListener('click', () => {
      const src = iframe.src.split('?')[0] + '?autoplay=true&muted=false';
      iframe.src = src;
      btn.style.display = 'none';
    });
  })();
